Под отсуствием понимать отсутсвие записи продукта в таблице товаров goods, а не нулевое количество

FROM brand LEFT OUTER JOIN product

SELECT
  warehouse.name,
  COUNT(*) AS Product,
  SUM(goods.quantity) AS units
FROM product
  INNER JOIN goods
    ON product.id = goods.producId
  INNER JOIN warehouse
    ON warehouse.ID = goods.worehouseId
GROUP BY warehouse.name
HAVING SUM(goods.quantity) > 100
